CREATE VIEW purchaseinfo AS SELECT firstname, lastname, description, quantitypurchased, costperitem, quantitypurchased*costperitem AS "Total purchase cost" FROM customer, product, purchase ORDER BY lastname;
SELECT * FROM purchaseinfo;
SELECT * FROM purchaseinfo WHERE quantitypurchased*costperitem >100;
